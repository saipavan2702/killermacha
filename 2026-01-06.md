// GCD Function
ll gcd(ll a, ll b) {
    if (b == 0) return a;
    return gcd(b, a % b);
}

// LCM Function
ll lcm(ll a, ll b) {
    return (a / gcd(a, b)) * b;
}

const int MAXN = 1e6;
vector<int> prime(MAXN + 1, 1);
vector<int> prime_nos;

void sieve() {
    prime[0] = 0;
    prime[1] = 0;
    for (int i = 2; i * i <= MAXN; i++) {
        for (int j = i * i; j <= MAXN; j += i) {
            if (prime[j] == 1) {
                prime[j] = 0;
            }
        }
    }
    for (int i = 2; i <= MAXN; i++) {
        if (prime[i] == 1) prime_nos.push_back(i);
    }
}
